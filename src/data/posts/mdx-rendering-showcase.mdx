### Python 代码

```python
# Python 代码示例
def fibonacci(n):
    """生成斐波那契数列"""
    if n <= 0:
        return []
    elif n == 1:
        return [0]
    elif n == 2:
        return [0, 1]

    fib = [0, 1]
    for i in range(2, n):
        fib.append(fib[i-1] + fib[i-2])
    return fib

# 使用示例
print(fibonacci(10))
```

### C++ 代码

```cpp
#include <iostream>
#include <vector>
#include <string>

class GameObject {
private:
    std::string name;
    int health;
    bool isActive;

public:
    GameObject(const std::string& objName, int hp)
        : name(objName), health(hp), isActive(true) {}

    void takeDamage(int damage) {
        health -= damage;
        if (health <= 0) {
            isActive = false;
            std::cout << name << " has been destroyed!" << std::endl;
        }
    }

    bool getIsActive() const { return isActive; }
};
```
